seccion: LucesNavidenas {

    mostrar("--- INICIANDO SECUENCIA NAVIDEÑA ---");
    
    // 1. Definimos las notas principales de Jingle Bells
    definir m = nota mi octava 4;
    definir s = nota sol octava 4;
    definir d = nota do octava 4;
    definir r = nota re octava 4;

    // 2. Definimos notas graves para hacer "bajeo" (simulando los destellos)
    definir doBajo = nota do octava 3;
    definir solBajo = nota sol octava 3;

    // 3. Estructuras musicales
    // Usamos los acordes como si fueran diferentes colores de luces encendiéndose
    acorde LuzRoja (doBajo, m, s);
    acorde LuzVerde (solBajo, r, s);
    
    // Las dos partes icónicas de la canción
    // Parte 1: "Navidad, Navidad"
    melodia JingleParte1 { m, m, m };
    
    // Parte 2: "Dulce Navidad"
    melodia JingleParte2 { m, s, d, r, m };

    // 4. Secuencia de parpadeo y sonido
    mostrar("Encendiendo arbolito...");
    
    // El ciclo simula la intermitencia de la serie navideña
    repetir 3 veces {
        
        // Destello 1 (Rojo)
        mostrar(" [*] LUZ ROJA ENCENDIDA [*] ");
        tocar(LuzRoja);
        tocar(JingleParte1);
        silencio;

        // Destello 2 (Verde)
        mostrar(" [*] LUZ VERDE ENCENDIDA [*] ");
        tocar(LuzVerde);
        tocar(JingleParte1);
        silencio;
        
        // Destello 3 (Dorado - Parte final de la estrofa)
        mostrar(" [*] LUZ DORADA DESTELLANDO [*] ");
        tocar(LuzRoja);
        tocar(JingleParte2);
        silencio;
    }
    
    mostrar("¡Feliz Navidad y Compilacion Exitosa!");

} fin
